SELECT *
FROM (
  SELECT *,
         ROW_NUMBER() OVER(ORDER BY _TABLE_SUFFIX DESC) AS row_num
  FROM `abc.analytics_255.event_*`
  WHERE _TABLE_SUFFIX >= FORMAT_DATE('%Y%m%d', DATE_SUB(CURRENT_DATE(), INTERVAL 1 MONTH))
    AND _TABLE_SUFFIX <= FORMAT_DATE('%Y%m%d', CURRENT_DATE())
)
WHERE row_num <= 30
